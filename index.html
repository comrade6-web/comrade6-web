// KONFIGURASI MESIN COMRADE6 const contractAddress = "0xC1e717690fF8B5820Af8C6a32ED6789805d4B366"; const usdtAddress = "0xc2132D05D31c914a87C6611C10748AEb04B58e8F"; // ABI (Bahasa Isyarat Kontrak) const contractABI = [ "function join(address _referrer, uint256 _package) external", "function users(address) view returns (bool, bool, address)", "function totalUsers() view returns (uint256)" ]; const usdtABI = [ "function approve(address spender, uint256 amount) external returns (bool)", "function balanceOf(address account) view returns (uint256)" ]; let provider, signer, contract, usdtContract; async function connectWallet() { if (!window.ethereum) return alert("Install MetaMask, Bro!"); provider = new ethers.BrowserProvider(window.ethereum); const accounts = await provider.send("eth_requestAccounts", []); const address = accounts[0]; signer = await provider.getSigner(); // Inisialisasi Kontrak contract = new ethers.Contract(contractAddress, contractABI, signer); usdtContract = new ethers.Contract(usdtAddress, usdtABI, signer); // Update UI Tombol const btn = document.querySelector('button'); // Tombol Navbar btn.innerText = address.substring(0, 6) + "..." + address.substring(38); btn.classList.replace('bg-black', 'bg-green-600'); updateDashboard(address); } async function updateDashboard(address) { const balance = await usdtContract.balanceOf(address); const formattedBalance = ethers.formatUnits(balance, 6); document.querySelector('h3.text-blue-600').innerText = "$" + parseFloat(formattedBalance).toFixed(2); const total = await contract.totalUsers(); document.querySelector('h3.text-4xl:not(.text-blue-600)').innerText = total.toString(); } async function joinSquad(amount) { if (!signer) return connectWallet(); const value = ethers.parseUnits(amount.toString(), 6); const referrer = "0x0000000000000000000000000000000000000000"; // Admin Root try { // Step 1: Izin USDT (Approve) alert("Langkah 1: Berikan izin USDT di MetaMask..."); const txApprove = await usdtContract.approve(contractAddress, value); await txApprove.wait(); // Step 2: Eksekusi Join alert("Langkah 2: Konfirmasi Join Squad..."); const txJoin = await contract.join(referrer, value); await txJoin.wait(); alert("SELAMAT! Kamu resmi bergabung di Squad!"); location.reload(); } catch (err) { console.error(err); alert("Transaksi Gagal, Bro. Cek Saldo USDT & POL kamu."); } } window.onload = () => { // Pasang Fungsi ke Tombol-tombol document.querySelectorAll('button').forEach(btn => { if (btn.innerText.includes("Connect Wallet")) btn.onclick = connectWallet; if (btn.innerText.includes("Mulai dari $6") || btn.innerText.includes("Join Squad - $6")) { btn.onclick = () => joinSquad(6); } if (btn.innerText.includes("Pilih Elite $60")) { btn.onclick = () => joinSquad(60); } }); };
